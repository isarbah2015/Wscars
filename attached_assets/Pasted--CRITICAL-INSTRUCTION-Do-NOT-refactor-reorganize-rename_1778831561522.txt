CHANGE 4 — app/auth/forgot-password.tsx
The file already has the correct logic. Only fix: verify that 
`services/firebase/auth.ts` exports both `sendPasswordResetEmail` and 
`authErrorMessage`. If that file does not exist, create it at 
`services/firebase/auth.ts` with exactly this content:

import { sendPasswordResetEmail as fbSendReset, getAuth } from 'firebase/auth';
import { auth } from '@/lib/firebase-persistence';

export async function sendPasswordResetEmail(email: string): Promise<void> {
  if (!auth) throw new Error('Auth not initialised');
  await fbSendReset(auth, email);
}

export function authErrorMessage(e: any): string {
  switch (e?.code) {
    case 'auth/invalid-email': return 'Invalid email address';
    case 'auth/user-not-found': return 'No account found with this email';
    case 'auth/too-many-requests': return 'Too many attempts. Try again later';
    default: return 'Something went wrong. Please try again';
  }
}
