🔧 Fix 1: Connect Google Login
Step 1: Install packages
bash
npm install @react-native-google-signin/google-signin @react-native-firebase/auth
Step 2: Update context/AppContext.tsx
Add imports:

tsx
import { GoogleSignin } from '@react-native-google-signin/google-signin';
import auth from '@react-native-firebase/auth';
Add configuration (inside the provider component):

tsx
useEffect(() => {
  GoogleSignin.configure({
    webClientId: process.env.EXPO_PUBLIC_GOOGLE_WEB_CLIENT_ID,
    offlineAccess: true,
  });
}, []);
Replace the stub handler:

tsx
const handleGoogleSignIn = async () => {
  try {
    await GoogleSignin.hasPlayServices();
    const { idToken } = await GoogleSignin.signIn();
    const googleCredential = auth.GoogleAuthProvider.credential(idToken);
    await auth().signInWithCredential(googleCredential);
    router.replace('/(tabs)/home');
  } catch (error) {
    console.error(error);
    alert(error.message);
  }
};

🔧 Fix 5: Premium Filter UI (Price Slider, Professional Look)
Current problem: Basic filter design, icons cut off, no range slider.
Solution – Use a slider for price range
Install a slider library:

bash
npm install @react-native-community/slider
Create a custom filter component (components/PriceSlider.tsx):

tsx
import Slider from '@react-native-community/slider';

export default function PriceSlider({ min, max, onValueChange }) {
  const [low, setLow] = useState(min);
  const [high, setHigh] = useState(max);
  return (
    <View style={{ padding: 16 }}>
      <Text style={styles.label}>Price range (GHS)</Text>
      <View style={{ flexDirection: 'row', justifyContent: 'space-between' }}>
        <Text>GHS {low}</Text>
        <Text>GHS {high}</Text>
      </View>
      <Slider
        minimumValue={min}
        maximumValue={max}
        value={high}
        onValueChange={val => {
          setHigh(val);
          onValueChange([low, val]);
        }}
        minimumTrackTintColor="#FF6B00"
        maximumTrackTintColor="#ddd"
        thumbTintColor="#FF6B00"
      />
      {/* For dual range, use two sliders or a library like 'multislider' */}
    </View>
  );
}
Improve filter layout (no cut‑off icons)
Add horizontal padding to scroll views: contentContainerStyle={{ paddingHorizontal: 16 }}

Ensure each filter chip has enough margin and no overflow:

tsx
<ScrollView horizontal showsHorizontalScrollIndicator={false} contentContainerStyle={{ paddingHorizontal: 16 }}>
  {brands.map(b => (
    <TouchableOpacity style={{ marginRight: 12, paddingVertical: 8, paddingHorizontal: 16, borderRadius: 30, backgroundColor: selected === b ? '#FF6B00' : '#f0f0f0' }}>
      <Text style={{ color: selected === b ? 'white' : '#333' }}>{b}</Text>
    </TouchableOpacity>
  ))}
</ScrollView>
Remove any absolute positioning that may cause clipping.
