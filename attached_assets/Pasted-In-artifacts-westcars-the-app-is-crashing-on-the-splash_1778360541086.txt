tsximport React, { useEffect } from 'react';
import { View, Image, StyleSheet, Dimensions } from 'react-native';
import { useRouter } from 'expo-router';
import { onAuthStateChanged } from 'firebase/auth';
import { auth } from '../lib/firebase-persistence';

const { width, height } = Dimensions.get('window');

export default function SplashScreen() {
  const router = useRouter();

  useEffect(() => {
    let navigationDone = false;

    const navigate = (destination: string) => {
      if (navigationDone) return;
      navigationDone = true;
      try {
        router.replace(destination as any);
      } catch (e) {
        setTimeout(() => {
          try { router.replace(destination as any); } catch (e2) {}
        }, 500);
      }
    };

    // Check auth state first then navigate
    const timeout = setTimeout(() => {
      const unsub = onAuthStateChanged(
        auth!,
        (user) => {
          unsub();
          navigate(user ? '/(tabs)' : '/auth/login');
        },
        () => {
          navigate('/auth/login');
        }
      );
    }, 2000);

    // Hard fallback — never stay on splash more than 6 seconds
    const hardFallback = setTimeout(() => {
      navigate('/auth/login');
    }, 6000);

    return () => {
      clearTimeout(timeout);
      clearTimeout(hardFallback);
    };
  }, [router]);

  return (
    <View style={styles.container}>
      <Image
        source={require('../assets/images/wc-logo.png')}
        style={styles.logo}
        resizeMode="contain"
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#0A1628',
    alignItems: 'center',
    justifyContent: 'center',
  },
  logo: {
    width: width * 0.6,
    height: height * 0.2,
  },
});

STEP 3 — Fix _layout.tsx Completely
Completely rewrite app/_layout.tsx from scratch with this exact production-safe pattern:
tsximport React, { useEffect, useRef, useState } from 'react';
import { Stack } from 'expo-router';
import * as SplashScreen from 'expo-splash-screen';
import { useFonts } from 'expo-font';
import { GestureHandlerRootView } from 'react-native-gesture-handler';
import { Ionicons } from '@expo/vector-icons';
import { Feather } from '@expo/vector-icons';
import { MaterialIcons } from '@expo/vector-icons';
import { MaterialCommunityIcons } from '@expo/vector-icons';
import { FontAwesome } from '@expo/vector-icons';
import { FontAwesome5 } from '@expo/vector-icons';
import { AntDesign } from '@expo/vector-icons';
import { Entypo } from '@expo/vector-icons';

// Prevent native splash from hiding until we are ready
SplashScreen.preventAutoHideAsync().catch(() => {});

export default function RootLayout() {
  const [appReady, setAppReady] = useState(false);
  const safetyFired = useRef(false);

  const [fontsLoaded, fontError] = useFonts({
    ...Ionicons.font,
    ...Feather.font,
    ...MaterialIcons.font,
    ...MaterialCommunityIcons.font,
    ...FontAwesome.font,
    ...FontAwesome5.font,
    ...AntDesign.font,
    ...Entypo.font,
  });

  // 5 second hard safety net — splash NEVER hangs forever
  useEffect(() => {
    const safety = setTimeout(() => {
      if (!safetyFired.current) {
        safetyFired.current = true;
        SplashScreen.hideAsync().catch(() => {});
        setAppReady(true);
      }
    }, 5000);
    return () => clearTimeout(safety);
  }, []);

  // Hide splash as soon as fonts resolve
  useEffect(() => {
    if (fontsLoaded || fontError) {
      if (!safetyFired.current) {
        safetyFired.current = true;
        SplashScreen.hideAsync().catch(() => {});
      }
      setAppReady(true);
    }
  }, [fontsLoaded, fontError]);

  if (!appReady) return null;

  return (
    <GestureHandlerRootView style={{ flex: 1 }}>
      <Stack screenOptions={{ headerShown: false }}>
        <Stack.Screen name="splash" />
        <Stack.Screen name="(tabs)" />
        <Stack.Screen name="auth/login" />
        <Stack.Screen name="auth/signup" />
        <Stack.Screen name="auth/forgot-password" />
        <Stack.Screen name="auth/phone" />
      </Stack>
    </GestureHandlerRootView>
  );
}
